152 /* Any write to the filehandle is appended to end of the file */
153 #define SPIFFS_APPEND (1<<0)
154 #define SPIFFS_O_APPEND SPIFFS_APPEND
155 /* If the opened file exists, it will be truncated to zero length before opened */
156 #define SPIFFS_TRUNC (1<<1)
157 #define SPIFFS_O_TRUNC SPIFFS_TRUNC
158 /* If the opened file does not exist, it will be created before opened */
159 #define SPIFFS_CREAT (1<<2)
160 #define SPIFFS_O_CREAT SPIFFS_CREAT
161 /* The opened file may only be read */
162 #define SPIFFS_RDONLY (1<<3)
163 #define SPIFFS_O_RDONLY SPIFFS_RDONLY
164 /* The opened file may only be written */
165 #define SPIFFS_WRONLY (1<<4)
166 #define SPIFFS_O_WRONLY SPIFFS_WRONLY
167 /* The opened file may be both read and written */
168 #define SPIFFS_RDWR (SPIFFS_RDONLY | SPIFFS_WRONLY)
169 #define SPIFFS_O_RDWR SPIFFS_RDWR
170 /* Any writes to the filehandle will never be cached but flushed directly */
171 #define SPIFFS_DIRECT (1<<5)
172 #define SPIFFS_O_DIRECT SPIFFS_DIRECT
173 /* If SPIFFS_O_CREAT and SPIFFS_O_EXCL are set, SPIFFS_open() shall fail if the file exists */
174 #define SPIFFS_EXCL (1<<6)
